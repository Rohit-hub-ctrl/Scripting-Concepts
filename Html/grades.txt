<html>
<body>
    <script>
        var marks = parseInt(prompt("Enter your marks"));
        if (marks >= 90) {
            document.write("Grade: A+");
        }
        else if (marks >= 70) {
            document.write("Grade: A");
        }
        else if (marks >= 60) {
            document.write("Grade: B");
        }
        else if (marks >= 40) {
            document.write("Grade: C");
        }
        else {
            document.write("Grade: Fail");
        }
    </script>
</body>
</html>
