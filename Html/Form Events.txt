  <script>
    document.getElementById("dropdown").addEventListener("change", function() {
      alert("You selected: " + this.value);
    });

    function addFocusBlurEvents(inputId, msgId, fieldName) {
      let input = document.getElementById(inputId);
      let msg = document.getElementById(msgId);

      input.addEventListener("focus", function() {
        msg.textContent = "Cursor entered the " + fieldName + " field!";
      });

      input.addEventListener("blur", function() {
        msg.textContent = "Cursor left the " + fieldName + " field!";
      });
    }

    addFocusBlurEvents("nameInput", "message", "Name");
    addFocusBlurEvents("email", "emailMsg", "Email");
    addFocusBlurEvents("password", "passMsg", "Password");
    addFocusBlurEvents("address", "addrMsg", "Address");
  </script>
